```cpp
#include<bits/stdc++.h>

using namespace std;

int n;

int main() {
    cin >> n;
    vector<int> love(n);

    // love[i]=>j

    // [i->j] is a directional graph line

    for (int i = 0; i < n; ++i) {
        cin >> love[i];
    }

    /*
     *  Case:3 1 2
     *  [1 -> 3] [A -> B (love[A-1])]
     *  [3 -> 2] [B -> C (love[B-1])]
     *  [2 -> 1] [C -> love[C-1]]
     */

    // Iterate over the whole list
    int A, B, C;
    for (int i = 1; i <= n; ++i) {
        A = i;
        B = love[A - 1];
        C = love[B - 1];

        if (love[C - 1] == A && A != C && B != C) {
            cout<< "YES" << endl;
            return 0;
        }
    }

    cout << "NO" << endl;
    return 0;
}
```
