https://codeforces.com/problemset/problem/996/A
```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int n;
    cin>>n;

    int counter{0};
    int cur{0};
    vector<int> bills{100,20,10,5};

    while(n>=5){
        if (n>=bills[cur]){
            n=n-bills[cur];
            counter++;
        }

        if (n<bills[cur]) cur++;
    }
    
    counter += n;

    cout<<counter<<endl;

    return 0;
}
```

One-pass solution.

On x86_64, it only require 2*4 bytes.

```cpp
#include <bits/stdc++.h>
using namespace std;
#define all(x) (x).begin(), (x).end()

int main(){
    int n;
    cin>>n;
    //1 5 10 20 100

    int counter{0};

    if(n/100)
        counter+=(n / 100), n-= (n / 100) * 100;

    if(n/20)
        counter+=(n / 20), n-= (n / 20) * 20;

    if(n/10)
        counter+=(n / 10), n-= (n / 10) * 10;

    if(n/5)
        counter+=(n / 5), n-= (n / 5) * 5;

    counter+=n;
    cout << counter << endl;

    return 0;
}
```
