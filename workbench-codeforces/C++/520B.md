BFS solution
```cpp
#include<bits/stdc++.h>
using namespace std;


bool calculated[1000001]={false};
long long counter[1000001]={0};

int main() {
    long long
        n, // => original
        m; // => desired

    // queue<T> is
    //    First in  => push
    // Access front => front
    //    First out => pop
    queue<long long> q;

    // In theory, we are dealing a tree
    // 1
    // |\
    // 2  3
    // |  \  \
    // 4  5   6

    // In practice, if we use Breadth-first search, we actually just
    // calculate each number as 1-(2-3)-(4-5-6)

    cin >> n >> m;

    q.push(n);
    calculated[n] = true;

    long long i;
    long long temp;
    while (!q.empty()) {

        //In each loop, we calculate 1 element from the whole thing.
        i = q.front();
        q.pop();

        temp=i*2;
        if (!calculated[temp] && temp <= m * 2 )
        {
            calculated[temp] = true;
            q.push(temp);
            counter[temp] = counter[i] + 1;
        }

        temp=i-1;
        if (!calculated[temp] && temp >= 0)
        {
            calculated[temp] = true;
            q.push(temp);
            counter[temp] = counter[i] + 1;
        }

        if (calculated[m])
            break;
    }

    cout << counter[m] << endl;

    return 0;
}
}
```
